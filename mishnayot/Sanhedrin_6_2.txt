הָיָה רָחוֹק מִבֵּית הַסְּקִילָה כְּעֶשֶׂר אַמּוֹת, אוֹמְרִים לוֹ הִתְוַדֵּה, שֶׁכֵּן דֶּרֶךְ הַמּוּמָתִין מִתְוַדִּין, שֶׁכָּל הַמִּתְוַדֶּה יֶשׁ לוֹ חֵלֶק לָעוֹלָם הַבָּא. שֶׁכֵּן מָצִינוּ בְעָכָן שֶׁאָמַר לוֹ יְהוֹשֻׁעַ, בְּנִי שִׂים נָא כָבוֹד לַה' אֱלֹהֵי יִשְׂרָאֵל וְתֶן לוֹ תוֹדָה וְגוֹ' וַיַּעַן עָכָן אֶת יְהוֹשֻׁעַ וַיֹּאמַר אָמְנָה אָנֹכִי חָטָאתִי לַה' אֱלֹהֵי יִשְׂרָאֵל וְכָזֹאת וְגוֹ' (יהושע ז). וּמִנַּיִן שֶׁכִּפֶּר לוֹ וִדּוּיוֹ, שֶׁנֶּאֱמַר (שם) וַיֹּאמֶר יְהוֹשֻׁעַ מֶה עֲכַרְתָּנוּ יַעְכָּרְךָ ה' בַּיּוֹם הַזֶּה. הַיּוֹם הַזֶּה אַתָּה עָכוּר, וְאִי אַתָּה עָכוּר לָעוֹלָם הַבָּא. וְאִם אֵינוֹ יוֹדֵעַ לְהִתְוַדּוֹת, אוֹמְרִים לוֹ, אֱמֹר תְּהֵא מִיתָתִי כַפָּרָה עַל כָּל עֲוֹנוֹתָי. רַבִּי יְהוּדָה אוֹמֵר, אִם הָיָה יוֹדֵעַ שֶׁהוּא מְזֻמָּם, אוֹמֵר תְּהֵא מִיתָתִי כַּפָּרָה עַל כָּל עֲוֹנוֹתַי חוּץ מֵעָוֹן זֶה. אָמְרוּ לוֹ, אִם כֵּן, יְהוּ כָל אָדָם אוֹמְרִים כָּךְ כְּדֵי לְנַקּוֹת אֶת עַצְמָן:
